Here's an example of code syntax highlighting:

class Animal {
  constructor(sound) {
    this.sound = sound;
  };

  speak(repeat = 1) {
    return this.sound.repeat(repeat);
  };
}

const dog = new Animal('bork');
console.log(dog.speak(5));
